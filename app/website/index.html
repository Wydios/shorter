Example Request


                    fetch("https://s.wydios.de/documents", {
                        method: "POST",
                        headers: {
                            "Content-Type": "application/json",
                            "Authorization": "Bearer IAmASecretPassword"
                        },
                        body: JSON.stringify({
                            username: "IAmACoolUser",
                            url: "https://twitch.tv/wydios",
                            days: 7 // Max Days is 7
                        })
                    });
